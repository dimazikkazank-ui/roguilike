app {
    выводит меню для выбора
    после выбора вызывает основной цикл game_cycle
}

datalayer {
    работа с сохранениями и статистикой
}

domain {
    characters {
        characters_fight {
            боевка персонажей
        }
        characters_movement {
            файл, описывающий интерфейс взаимодействия персонажей и окружения
            В данном файле описаны передвижения монстров по комнате, передвижение
            игрока по комнате, а также взаимодействие игрока с противниками 
            в смысле их уровня враждебности
        }
        consumables_actions {
            В данном файле описаны механики подбора, потребления расходников
        }
    }
    data_structs {
        dsu {
            какие-то функции про множества для генерации и объединения комнат
        }
        graph {
            dist_and_next_pos_to_target() {
                ищет кратчайший путь от монстра до игрока с учетом стен
                возвращает массив направлений движения
                лучше возвращать одно направление, т к каждый кадр будет заново меняться массив
            }
        }
        queue {
            поиск пути монстра к игроку

        }
        vector {
            структура для хранения пути от монстра до игрока
        }
    }
    entities {
        куча структур и констант под сущности
    }
    generation {
        generation {
            генерация комнат и расходников
        }
        random {
            рандом и перестановка массива
        }
        tools {
            min max
        }
    }
}

presentation {
    controller {
        Проверки состояний:
        check_death() - проверяет смерть игрока
        check_end() - проверяет завершение игры
        check_level_end() - проверяет завершение уровня

        Меню инвентаря:
        print_weapon_menu() - меню выбора оружия
        print_food_menu() - меню выбора еды
        print_scroll_menu() - меню выбора свитков
        print_elixir_menu() - меню выбора эликсиров

        Боевая система:
        attack_monster_ui() - отображение атаки на монстра
        process_player_move_ui() - обработка хода игрока
        process_monsters_move_ui() - обработка хода монстров
        print_data_about_monster_attack() - отображение атаки монстра

        Управление и ввод:
        process_user_input() - обработка пользовательского ввода:
        WASD - движение
        H - выбор оружия
        J - использование еды
        K - использование эликсиров
        E - использование свитков
        ESC - выход

        Игровой цикл (game_cycle):
        Отображение карты и интерфейса
        Обработка ввода пользователя
        Проверка условий:
        Завершение уровня
        Смерть игрока
        Конец игры
        Сохранение прогресса
    }
    presentation {
        рендеринг
        get_room_by_coord() - находит комнату по координатам
        on_the_same_room_or_passage() - монстр видим если в той же зоне
        check_unique() - монстр в бою всегда видим
    }
    screen {
        экраны смерти, начала и концы
    }
}
